\tikzset{
  pback/.style={draw=#1!55!black, fill=#1!18, rounded corners=2pt,
                minimum width=2.0cm, minimum height=2.5cm, line width=0.4pt},
  pfront/.style={draw=#1!70!black, fill=#1!7, rounded corners=2pt,
                 minimum width=2.0cm, minimum height=2.5cm, line width=0.8pt},
  flow/.style={-{Stealth[length=2.6mm]}, line width=1.0pt, draw=cgray},
  hdr/.style={font=\sffamily\bfseries\small, text=#1!55!black, align=center},
  lbl/.style={font=\sffamily\small, align=center, text=#1!50!black},
  cnt/.style={font=\sffamily\bfseries\normalsize, text=black},
  procbox/.style={draw=#1!70!black, fill=#1!9, rounded corners=5pt,
                  line width=1.0pt, align=center, font=\sffamily,
                  inner sep=8pt},
}

% --- a three-page "stack of renders"; front node is named #1 --------------
% #1 name   #2 x   #3 y   #4 color
\newcommand{\pagestack}[4]{%
  \node[pback=#4] at ($(#2,#3)+(0.24,0.24)$) {};
  \node[pback=#4] at ($(#2,#3)+(0.12,0.12)$) {};
  \node[pfront=#4] (#1) at (#2,#3) {};
}
% --- tiny anatomical glyphs drawn on a page centred at (#1,#2) -------------
\newcommand{\spineicon}[3]{% x y color
  \foreach \i in {0,1,2,3,4}{%
    \fill[#3!35!white, draw=#3!70!black, line width=0.3pt, rounded corners=1pt]
      ($(#1,#2)+(-0.26, 0.82-0.40*\i)$) rectangle ($(#1,#2)+(0.26, 1.10-0.40*\i)$);}}
\newcommand{\pelvisicon}[3]{% x y color
  \fill[#3!45!white, draw=#3!70!black, line width=0.3pt]
    ($(#1,#2)+(0,0.45)$) -- ($(#1,#2)+(-0.32,-0.15)$) -- ($(#1,#2)+(0.32,-0.15)$) -- cycle;
  \fill[#3!30!white, draw=#3!70!black, line width=0.3pt, rotate around={22:($(#1,#2)+(-0.55,0)$)}]
    ($(#1,#2)+(-0.82,-0.45)$) rectangle ($(#1,#2)+(-0.30,0.45)$);
  \fill[#3!30!white, draw=#3!70!black, line width=0.3pt, rotate around={-22:($(#1,#2)+(0.55,0)$)}]
    ($(#1,#2)+(0.30,-0.45)$) rectangle ($(#1,#2)+(0.82,0.45)$);}
\newcommand{\ribsicon}[3]{% x y color  (rib arcs, both sides)
  \foreach \i in {0,1,2}{%
    \draw[#3!75!black, line width=0.5pt]
      ($(#1,#2)+(-0.05,1.05-0.30*\i)$) to[bend left=45] ($(#1,#2)+(-0.95,0.75-0.30*\i)$);
    \draw[#3!75!black, line width=0.5pt]
      ($(#1,#2)+(0.05,1.05-0.30*\i)$) to[bend right=45] ($(#1,#2)+(0.95,0.75-0.30*\i)$);}}
